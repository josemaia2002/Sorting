227template <typename T>
228void shell(T *first, T *last) {
229 int tamanho = last - first;
230 // Comece com um gap grande e, em seguida, reduz o gap até que ele se torne 1
231 for (int gap = tamanho / 2; gap > 0; gap /= 2) {
232 // Realize a ordenação por inserção em cada sublista definida pelo gap
233 for (int i = gap; i < tamanho; i++) {
234 int temp = first[i];
235 int j;
236 for (j = i; j >= gap && temp < first[j - gap]; j -= gap) {
237 first[j] = first[j - gap];
238 }
239 first[j] = temp;
240 }
